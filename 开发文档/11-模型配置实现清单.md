### 1. API Integration in ModelConfigForm.vue

#### Save Configuration
```javascript
// Replace mock function in handleSave()
import { saveAiModelConfigApi } from '@/api/sysAiConfig'
import { ElMessage } from 'element-plus'

const handleSave = async () => {
  // Validate
  if (!formData.apiKey.trim()) {
    ElMessage.warning('Please enter an API Key')
    return
  }
  if (!formData.modelId.trim()) {
    ElMessage.warning('Please enter a Model ID')
    return
  }

  isSaving.value = true
  try {
    const { data } = await saveAiModelConfigApi({
      model_type: formData.modelId,
      api_key: formData.apiKey,
      base_url: formData.baseUrl || null,
      context_length: formData.contextLength || null,
      custom_headers: formData.customHeaders.filter(h => h.key && h.value)
    })
    
    if (data?.code === 200) {
      ElMessage.success(data.msg || 'Configuration saved successfully')
      formData.apiKey = '' // Clear sensitive data
    } else {
      ElMessage.error(data?.msg || 'Failed to save configuration')
    }
  } catch (error) {
    const msg = error?.response?.data?.detail || 
                error?.response?.data?.msg || 
                'Failed to save configuration'
    ElMessage.error(msg)
  } finally {
    isSaving.value = false
  }
}
```

#### Test Connection
```javascript
// Replace mock function in handleTest()
import { testAiConnectionApi } from '@/api/assistant'
import { ElMessage } from 'element-plus'

const handleTest = async () => {
  // Validate
  if (!formData.apiKey.trim()) {
    ElMessage.warning('Please enter an API Key')
    return
  }
  if (!formData.modelId.trim()) {
    ElMessage.warning('Please enter a Model ID')
    return
  }

  isTesting.value = true
  try {
    const { data } = await testAiConnectionApi({
      api_key: formData.apiKey,
      model: formData.modelId,
      api_base_url: formData.baseUrl || undefined
    })
    
    if (data?.success) {
      const reply = data.reply
      ElMessage.success(
        reply ? `${data.message || 'Connection successful'}: ${reply}` 
              : data.message || 'Connection successful'
      )
    } else {
      ElMessage.error(data?.error || data?.message || 'Connection failed')
    }
  } catch (error) {
    const msg = error?.response?.data?.error || 
                error?.response?.data?.msg || 
                'Connection test failed'
    ElMessage.error(msg)
  } finally {
    isTesting.value = false
  }
}
```

#### Load Existing Configuration
```javascript
// Add to script setup
import { onMounted } from 'vue'
import { getAiModelConfigApi } from '@/api/sysAiConfig'

onMounted(async () => {
  try {
    const { data } = await getAiModelConfigApi()
    if (data?.code === 200 && data?.data) {
      formData.baseUrl = data.data.base_url || ''
      formData.modelId = data.data.model_type || ''
      formData.contextLength = data.data.context_length || null
      formData.customHeaders = data.data.custom_headers || []
      // Note: Never load API Key for security reasons
    }
  } catch (error) {
    console.error('Failed to load configuration:', error)
  }
})
```

## 📞 Quick Start Guide

### To Test the New Interface:

1. **Start the development server:**
   ```bash
   cd frontend
   npm run dev
   ```

2. **Navigate to the configuration page:**
   - Open browser: `http://localhost:5173`
   - Login to AITesta
   - Go to: 智能助手 → 模型接入 tab

3. **Test the interface:**
   - Fill in Base URL (e.g., `https://api.openai.com/v1`)
   - Enter API Key
   - Enter Model ID (e.g., `gpt-4o`)
   - Click eye icon to toggle password visibility
   - Expand "Advanced Configuration"
   - Add custom headers
   - Click "测试连接" to test
   - Click "保存配置" to save
