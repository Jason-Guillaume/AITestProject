### 2.1 基于需求文档的功能测试

#### 2.1.1 需求分析 → 测试用例生成

```python
# assistant/services/functional_test_generator.py

from assistant.services.llm_service import LLMService
from testcase.models import TestCase, TestCaseStep

class FunctionalTestGenerator:
    """AI 辅助功能测试生成器"""
    
    def __init__(self):
        self.llm = LLMService()
    
    async def generate_tests_from_requirements(self, requirements_text: str, module_id: int) -> dict:
        """根据需求文档生成测试用例"""
        
        prompt = f"""
        你是一个专业的测试工程师。请根据以下需求文档生成详细的测试用例：
        
        【需求文档】
        {requirements_text}
        
        请生成测试用例，包含：
        1. 测试用例标题
        2. 测试类型（功能/边界/异常/安全）
        3. 前置条件
        4. 测试步骤
        5. 预期结果
        6. 测试数据
        
        请以 JSON 格式返回：
        [
            {{"title": "用例标题", "type": "functional", "steps": [{"action": "步骤", "expected": "结果"}]}}
        ]
        
        请确保覆盖：正常流程、边界条件、异常情况、安全测试
        """
        
        response = await self.llm.generate(prompt, response_format="json")
        return response
    
    async def generate_api_tests_from_spec(self, api_spec: str) -> dict:
        """根据 API 规范生成 API 测试用例"""
        
        prompt = f"""
        你是一个 API 测试专家。请根据以下 API 规范生成 API 测试用例：
        
        【API 规范】
        {api_spec}
        
        请生成测试用例，包含：正常场景、参数验证、边界值、异常场景、安全测试
        返回 JSON 格式测试用例列表。
        """
        
        response = await self.llm.generate(prompt, response_format="json")
        return response
```

#### 2.1.2 测试用例存储

```python
# testcase/services/test_case_service.py

from testcase.models import TestCase, TestCaseStep
from django.utils import timezone

class TestCaseService:
    """测试用例服务"""
    
    @staticmethod
    async def create_from_ai_response(ai_response: dict, module_id: int, user_id: int):
        """从 AI 响应创建测试用例"""
        test_cases = []
        
        for case_data in ai_response.get("test_cases", []):
            test_case = TestCase.objects.create(
                title=case_data["title"],
                module_id=module_id,
                created_by_id=user_id,
                test_type=case_data.get("type", "functional"),
                priority=case_data.get("priority", "medium"),
                status="draft",
                create_time=timezone.now(),
                update_time=timezone.now()
            )
            
            for idx, step_data in enumerate(case_data.get("steps", []), 1):
                TestCaseStep.objects.create(
                    test_case=test_case,
                    step_number=idx,
                    description=step_data["action"],
                    expected_result=step_data["expected"]
                )
            
            test_cases.append(test_case)
        
        return test_cases
```

### 2.2 基于 API 文档的功能测试

```python
# execution/services/api_test_generator.py

import json
from execution.models import ApiScenario, ApiScenarioStep

class APITestGenerator:
    """API 测试生成器"""
    
    async def generate_from_openapi(self, openapi_spec: str) -> dict:
        """根据 OpenAPI 规范生成 API 测试场景"""
        
        prompt = f"""
        根据以下 OpenAPI 规范生成 API 测试场景：
        
        【OpenAPI 规范】
        {openapi_spec}
        
        返回 JSON 格式：
        [
            {{"scenario_name": "场景名称", "base_url": "https://api.example.com",
              "tests": [
                {{"name": "测试名称", "method": "GET", "path": "/api",
                  "request": {{"headers": {{}}, "body": {{}}}},
                  "assertions": [
                    {{"type": "status_code", "expected": 200}},
                    {{"type": "response_time", "max_ms": 500}}
                  ]}}]}}]
        """
        
        response = await self.llm.generate(prompt, response_format="json")
        return response
    
    @staticmethod
    def save_scenario_to_db(scenario_data: dict, user_id: int):
        """保存测试场景到数据库"""
        scenario = ApiScenario.objects.create(
            name=scenario_data["scenario_name"],
            base_url=scenario_data.get("base_url", ""),
            created_by_id=user_id,
            is_active=True
        )
        
        for test_data in scenario_data.get("tests", []):
            ApiScenarioStep.objects.create(
                scenario=scenario,
                step_number=1,
                method=test_data["method"],
                url=f"{scenario.base_url}{test_data['path']}",
                headers=json.dumps(test_data["request"].get("headers", {})),
                body=json.dumps(test_data["request"].get("body", {})),
                assertions=json.dumps(test_data.get("assertions", []))
            )
        
        return scenario
```

```python
# data/services/test_data_generator.py

from assistant.services.llm_service import LLMService
from faker import Faker

class TestDataGenerator:
    """智能测试数据生成器"""
    
    def __init__(self):
        self.llm = LLMService()
        self.faker = Faker('zh_CN')
    
    async def generate_from_schema(self, schema: dict, count: int = 10) -> list:
        """根据数据 Schema 生成测试数据"""
        
        prompt = f"""
        根据以下数据 Schema 生成 {count} 条测试数据：
        
        【Schema】
        {schema}
        
        请生成符合 Schema 的测试数据，包含正常数据和边界数据。
        返回 JSON 格式的数据列表。
        """
        
        response = await self.llm.generate(prompt, response_format="json")
        return response
    
    def generate_simple_data(self, data_type: str, count: int = 10) -> list:
        """生成简单测试数据"""
        data = []
        for _ in range(count):
            if data_type == "user":
                data.append({
                    "username": self.faker.user_name(),
                    "email": self.faker.email(),
                    "phone": self.faker.phone_number()
                })
            elif data_type == "order":
                data.append({
                    "product": self.faker.word(),
                    "quantity": self.faker.random_int(1, 100),
                    "price": self.faker.random_int(10, 1000)
                })
        return data
```

```python
# API 示例

from assistant.services.functional_test_generator import FunctionalTestGenerator

async def create_tests_from_requirements():
    generator = FunctionalTestGenerator()

    用户登录功能：

    3. 验证登录成功跳转到首页
```

## 六、总结

AI 辅助功能测试的核心流程：

1. **输入**：需求文档、API 文档、页面截图、用户操作录屏
2. **AI 分析**：需求理解、接口分析、元素识别、流程提取
3. **测试生成**：用例生成、数据生成、断言生成、脚本生成
4. **执行**：API 测试、UI 测试、端到端测试、回归测试

通过 AI 辅助，可以大幅减少手工编写测试用例的时间，提高测试覆盖率，实现快速测试迭代。
