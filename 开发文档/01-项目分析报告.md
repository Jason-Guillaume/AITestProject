# SmartTest（AITestProduct）项目分析报告

## 一、项目概述

SmartTest 是一个面向软件测试全流程的 AI 辅助平台，覆盖测试设计、用例管理、执行报告、缺陷跟踪与性能任务管理，并集成大模型能力用于测试用例生成与分析。

### 技术栈

| 层级 | 技术 |
|------|------|
| 后端 | Django + DRF + PostgreSQL/MySQL + Redis + Celery |
| 前端 | Vue 3 + Vite + Element Plus + Tailwind CSS + Vue Router + Pinia |
| 测试执行 | requests + k6 + JMeter + Locust |
| AI/RAG | OpenAI API + Ollama + Chroma 向量数据库 |
| 实时通信 | WebSocket (Daphne) + SSE |
| 部署 | Docker + Nginx + Gunicorn/Uvicorn |

### 4.3 后端执行引擎

```python
# execution/services/ui_executor.py

import asyncio
from playwright.async_api import async_playwright

class UIExecutor:
    """UI 自动化执行引擎"""
    
    BROWSER_TYPES = {"chromium": "chromium", "firefox": "firefox", "webkit": "webkit"}
    
    def __init__(self, base_url: str, browser_type: str = "chromium", headless: bool = True):
        self.base_url = base_url
        self.browser_type = browser_type
        self.headless = headless
        self.playwright = None
        self.browser = None
        self.context = None
        self.page = None
    
    async def start(self):
        self.playwright = await async_playwright().start()
        browser_type = self.BROWSER_TYPES.get(self.browser_type, "chromium")
        self.browser = await self.playwright[browser_type].launch(
            headless=self.headless,
            args=["--no-sandbox", "--disable-setuid-sandbox"]
        )
        self.context = await self.browser.new_context(viewport={"width": 1920, "height": 1080})
        self.page = await self.context.new_page()
        await self.page.set_default_timeout(30000)
        await self.page.set_default_navigation_timeout(30000)
    
    async def stop(self):
        if self.context: await self.context.close()
        if self.browser: await self.browser.close()
        if self.playwright: await self.playwright.stop()
    
    async def navigate_to(self, url: str):
        full_url = url if url.startswith(("http://", "https://")) else f"{self.base_url}{url}"
        await self.page.goto(full_url)
        await self.page.wait_for_load_state("networkidle")
    
    async def click(self, locator: str):
        await self.page.locator(locator).click()
    
    async def input_text(self, locator: str, text: str):
        await self.page.locator(locator).fill(text)
    
    async def assert_visible(self, locator: str):
        await self.page.locator(locator).wait_for(state="visible")
    
    async def assert_text_equals(self, locator: str, expected: str):
        actual = await self.page.locator(locator).text_content()
        assert actual == expected, f"Expected '{expected}', got '{actual}'"
    
    async def take_screenshot(self, path: str):
        await self.page.screenshot(path=path, full_page=True)
    
    async def execute_steps(self, steps: list):
        """执行步骤列表"""
        logs = []
        error_message = ""
        status = "success"
        
        for idx, step in enumerate(steps, 1):
            try:
                log = {"step": idx, "action": step["action_type"], "status": "running"}
                
                if step["action_type"] == "navigate":
                    await self.navigate_to(step["value"])
                elif step["action_type"] == "click":
                    await self.click(step["locator"])
                elif step["action_type"] == "input":
                    await self.input_text(step["locator"], step["value"])
                elif step["action_type"] == "assert":
                    if step.get("assertion_type") == "visible":
                        await self.assert_visible(step["locator"])
                    elif step.get("assertion_type") == "text_equals":
                        await self.assert_text_equals(step["locator"], step["expected_result"])
                
                log["status"] = "success"
                logs.append(log)
                
            except Exception as e:
                log["status"] = "failed"
                log["error"] = str(e)
                logs.append(log)
                error_message = str(e)
                status = "failed"
                break
        
        return {"logs": logs, "status": status, "error_message": error_message}
```

```python
# assistant/services/ui_automation_generator.py

from assistant.services.llm_service import LLMService

class UIAutomationGenerator:
    """AI 辅助 UI 自动化脚本生成器"""
    
    def __init__(self):
        self.llm = LLMService()
    
    async def generate_from_description(self, description: str, page_url: str) -> dict:
        """根据自然语言描述生成 UI 自动化步骤"""
        
        prompt = f"""
        你是一个 UI 自动化测试专家。请根据以下需求生成 Playwright 测试脚本步骤：
        
        页面 URL: {page_url}
        测试需求：{description}
        
        请返回 JSON 格式的步骤列表：
        [
            {{
                "action_type": "navigate|click|input|assert|screenshot",
                "description": "步骤描述",
                "locator": "元素定位符",
                "locator_type": "css|xpath|id|name|text|data-testid",
                "value": "操作值（可选）",
                "expected_result": "预期结果（可选）",
                "assertion_type": "visible|text_equals|attribute_equals（可选）"
            }}
        ]
        """
        
        response = await self.llm.generate(prompt, response_format="json")
        return response
    
    async def generate_from_screenshot(self, screenshot_base64: str, description: str) -> dict:
        """根据页面截图和描述生成测试步骤"""
        
        prompt = f"""
        你看到一个网页截图。请根据截图中的元素和以下需求生成测试步骤：
        
        测试需求：{description}
        
        请识别页面中的关键元素（按钮、输入框等），并生成相应的测试步骤。
        返回 JSON 格式的步骤列表。
        """
        
        response = await self.llm.generate_with_image(prompt, screenshot_base64, response_format="json")
        return response
    
    async def optimize_locator(self, locator: str, page_html: str) -> dict:
        """优化元素定位符，生成更稳定的定位策略"""
        
        prompt = f"""
        请优化以下元素定位符，使其更稳定可靠：
        
        当前定位符：{locator}
        页面 HTML 片段：{page_html[:2000]}
        
        请返回优化后的定位符，优先使用 data-testid 或稳定的 CSS 选择器。
        返回 JSON 格式：{{"optimized_locator": "...", "locator_type": "...", "confidence": 0.9}}
        """
        
        response = await self.llm.generate(prompt, response_format="json")
        return response
```
